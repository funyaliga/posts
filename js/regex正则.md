### 密码 ###

```javascript
/^(?=.*\d)(?=.*[a-z])(?=.*[A-Z])[0-9a-zA-Z]{8,}$/

/^
  (?=.*\d)          // 至少一个数字
  (?=.*[a-z])       // 至少一个小写英语
  (?=.*[A-Z])       // 至少一个大写英语
  [a-zA-Z0-9]{8,16} // 8到16位
$/
```



### email

```javascript
/^[^\s@]+@[^\s@]+\.[^\s@]+$/

/^
  [^\s@]+     // 非空白字符和@符号
  @           // @符号
  [^\s@]+     // 非空白字符和@符号
  \.          // .
  [^\s@]+     // 非空白字符和@符号
$/
```



### 手机号 ###

```javascript
/^1[3-9]\d{9}$/

/^
  1      // 1开头
  [3-9]  // 3-9号码段
  \d{9}  // 后面9位
$/
```





### 身份证 ###

![身份证](/Users/junli/work/fun/posts/img/身份证.jpeg)

```javascript
// 简陋版
/(^[1-9]\d{5}(18|19|([23]\d))\d{2}((0[1-9])|(10|11|12))(([0-2][1-9])|10|20|30|31)\d{3}[0-9Xx]$)|(^[1-9]\d{5}\d{2}((0[1-9])|(10|11|12))(([0-2][1-9])|10|20|30|31)\d{3}$)/


/
  (^                             // ↓↓↓ 18位身份证 ↓↓↓
    [1-9]\d{5}                   // 前六位地区，非0打头
    (18|19|([23]\d))\d{2}        // 出身年份，覆盖范围为 1800-3999 年
    ((0[1-9])|(10|11|12))        // 月份，01-12月
    (([0-2][1-9])|10|20|30|31)   // 日期，01-31天
    \d{3}[0-9Xx]                 // 顺序码三位 + 一位校验码
  $)
  |
  (^                             // ↓↓↓ 15位身份证 ↓↓↓
    [1-9]\d{5}                   // 前六位地区，非0打头  
    \d{2}((0[1-9])|(10|11|12))   // 月份，01-12月
    (([0-2][1-9])|10|20|30|31)   // 日期，01-31天
    \d{3}                        // 顺序码三位，没有校验码
  $)
/

```
